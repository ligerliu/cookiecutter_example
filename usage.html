Example¶

In [1]: import numpy as np

In [2]: from pkg_resources import resource_filename

In [3]: path = resource_filename('package_example','example_data/example_data.npz')

In [4]: a = np.load(path)

In [5]: saxs = a['example_data'].item()['saxs']

In [6]: reflectivity = a['example_data'].item()['reflectivity']

In [7]: transmission = a['example_data'].item()['transmission']

In [8]: q_reflc = a['example_data'].item()['q_reflc']

In [9]: SLD = a['example_data'].item()['SLD']

In [10]: SLDS = a['example_data'].item()['SLDS']

In [11]: incident_angle = a['example_data'].item()['incident_angle']

In [12]: wavelength = 0.9184

In [13]: detector_distance = 1.2

In [14]: pixel_size = 75*1e-6

In [15]: scale_factor = 10

In [16]: beamcenter_y = 256
